332 template<class KmeansMatrix_ = kmeans::SimpleMatrix<Index_, Data_> >
333 KmknnPrebuilt(
334 std::size_t num_dim,
335 Index_ num_obs,
336 std::vector<Common<Data_, Distance_> > data,
337 std::shared_ptr<const DistanceMetric_> metric,
338 const KmknnOptions<Index_, Data_, Distance_, KmeansMatrix_>& options
339 ) :
340 my_dim(num_dim),
341 my_obs(num_obs),
342 my_data(std::move(data)),
343 my_metric(std::move(metric))
344 {
345 auto init = options.initialize_algorithm;
346 if (init == nullptr) {
347 init.reset(new InitializeKmeanspp<Index_, Data_, Distance_, KmeansMatrix_>);
348 }
349 auto refine = options.refine_algorithm;
350 if (refine == nullptr) {
351 refine.reset(new RefineHartiganWong<Index_, Data_, Distance_, KmeansMatrix_>);
352 }
353
354 Index_ ncenters = std::ceil(std::pow(my_obs, options.power));
355 my_centers.resize(static_cast<std::size_t>(ncenters) * my_dim); // cast to avoid overflow problems.
356
357 kmeans::SimpleMatrix<Index_, Data_> mat(my_dim, my_obs, my_data.data());
358 std::vector<Index_> clusters(my_obs);
359 auto output = kmeans::compute(mat, *init, *refine, ncenters, my_centers.data(), clusters.data());
360
361 // Removing empty clusters, e.g., due to duplicate points.
362 {
363 my_sizes.resize(ncenters);
364 std::vector<Index_> remap(ncenters);
365 Index_ survivors = 0;
366 for (Index_ c = 0; c < ncenters; ++c) {
367 if (output.sizes[c]) {
368 if (c > survivors) {
369 auto src = my_centers.begin() + static_cast<std::size_t>(c) * my_dim; // cast to avoid overflow.
370 auto dest = my_centers.begin() + static_cast<std::size_t>(survivors) * my_dim;
371 std::copy_n(src, my_dim, dest);
372 }
373 remap[c] = survivors;
374 my_sizes[survivors] = output.sizes[c];
375 ++survivors;
376 }
377 }
378
379 if (survivors < ncenters) {
380 for (auto& c : clusters) {
381 c = remap[c];
382 }
383 ncenters = survivors;
384 my_centers.resize(static_cast<std::size_t>(ncenters) * my_dim);
385 my_sizes.resize(ncenters);
386 }
387 }
388
389 my_offsets.resize(ncenters);
390 for (Index_ i = 1; i < ncenters; ++i) {
391 my_offsets[i] = my_offsets[i - 1] + my_sizes[i - 1];
392 }
393
394 // Organize points correctly; firstly, sorting by distance from the assigned center.
395 std::vector<std::pair<Distance_, Index_> > by_distance(my_obs);
396 {
397 auto sofar = my_offsets;
398 auto host = my_data.data();
399 for (Index_ o = 0; o < my_obs; ++o) {
400 auto optr = host + static_cast<std::size_t>(o) * my_dim;
401 auto clustid = clusters[o];
402 auto cptr = my_centers.data() + static_cast<std::size_t>(clustid) * my_dim;
403
404 auto& counter = sofar[clustid];
405 auto& current = by_distance[counter];
406 current.first = my_metric->normalize(my_metric->raw(my_dim, optr, cptr));
407 current.second = o;
408
409 ++counter;
410 }
411
412 for (Index_ c = 0; c < ncenters; ++c) {
413 auto begin = by_distance.begin() + my_offsets[c];
414 std::sort(begin, begin + my_sizes[c]);
415 }
416 }
417
418 // Permuting in-place to mirror the reordered distances, so that the search is more cache-friendly.
419 {
420 auto host = my_data.data();
421 std::vector<uint8_t> used(my_obs);
422 std::vector<Data_> buffer(my_dim);
423 my_observation_id.resize(my_obs);
424 my_dist_to_centroid.resize(my_obs);
425 my_new_location.resize(my_obs);
426
427 for (Index_ o = 0; o < my_obs; ++o) {
428 if (used[o]) {
429 continue;
430 }
431
432 const auto& current = by_distance[o];
433 my_observation_id[o] = current.second;
434 my_dist_to_centroid[o] = current.first;
435 my_new_location[current.second] = o;
436 if (current.second == o) {
437 continue;
438 }
439
440 // We recursively perform a "thread" of replacements until we
441 // are able to find the home of the originally replaced 'o'.
442 auto optr = host + static_cast<std::size_t>(o) * my_dim;
443 std::copy_n(optr, my_dim, buffer.begin());
444 Index_ replacement = current.second;
445 do {
446 auto rptr = host + static_cast<std::size_t>(replacement) * my_dim;
447 std::copy_n(rptr, my_dim, optr);
448 used[replacement] = 1;
449
450 const auto& next = by_distance[replacement];
451 my_observation_id[replacement] = next.second;
452 my_dist_to_centroid[replacement] = next.first;
453 my_new_location[next.second] = replacement;
454
455 optr = rptr;
456 replacement = next.second;
457 } while (replacement != o);
458
459 std::copy(buffer.begin(), buffer.end(), optr);
460 }
461 }
462
463 return;
464 }

467 void search_nn(const Common<Data_, Distance_>* target, knncolle::NeighborQueue<Index_, Distance_>& nearest, std::vector<std::pair<Distance_, Index_> >& center_order) const {
468 /* Computing distances to all centers and sorting them. The aim is to
469 * go through the nearest centers first, to try to get the shortest
470 * threshold (i.e., 'nearest.limit()') possible at the start;
471 * this allows us to skip searches of the later clusters.
472 */
473 center_order.clear();
474 std::size_t ncenters = my_sizes.size();
475 center_order.reserve(ncenters);
476 auto clust_ptr = my_centers.data();
477 for (std::size_t c = 0; c < ncenters; ++c, clust_ptr += my_dim) {
478 center_order.emplace_back(my_metric->raw(my_dim, target, clust_ptr), c);
479 }
480 std::sort(center_order.begin(), center_order.end());
481
482 // Computing the distance to each center, and deciding whether to proceed for each cluster.
483 Distance_ threshold_raw = std::numeric_limits<Distance_>::infinity();
484 for (const auto& curcent : center_order) {
485 const Index_ center = curcent.second;
486 const Distance_ dist2center = my_metric->normalize(curcent.first);
487
488 const auto cur_nobs = my_sizes[center];
489 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
490 const Distance_ maxdist = *(dIt + cur_nobs - 1);
491
492 Index_ firstcell = 0;
493#if KNNCOLLE_KMKNN_USE_UPPER
494 Distance_ upper_bd = std::numeric_limits<Distance_>::max();
495#endif
496
497 if (!std::isinf(threshold_raw)) {
498 const Distance_ threshold = my_metric->normalize(threshold_raw);
499
500 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
501 * threshold + maxdist < dist2center
502 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
503 */
504 const Distance_ lower_bd = dist2center - threshold;
505 if (maxdist < lower_bd) {
506 continue;
507 }
508
509 firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
510
511#if KNNCOLLE_KMKNN_USE_UPPER
512 /* This exploits the reverse triangle inequality, to ignore points where:
513 * threshold + dist2center < point-to-center distance
514 */
515 upper_bd = threshold + dist2center;
516#endif
517 }
518
519 const auto cur_start = my_offsets[center];
520 const auto* other_cell = my_data.data() + my_dim * static_cast<std::size_t>(cur_start + firstcell); // cast to avoid overflow issues.
521 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_cell += my_dim) {
522#if KNNCOLLE_KMKNN_USE_UPPER
523 if (*(dIt + celldex) > upper_bd) {
524 break;
525 }
526#endif
527
528 auto dist2cell_raw = my_metric->raw(my_dim, target, other_cell);
529 if (dist2cell_raw <= threshold_raw) {
530 nearest.add(cur_start + celldex, dist2cell_raw);
531 if (nearest.is_full()) {
532 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
533#if KNNCOLLE_KMKNN_USE_UPPER
534 upper_bd = my_metric->normalize(threshold_raw) + dist2center;
535#endif
536 }
537 }
538 }
539 }

543 void search_all(const Data_* target, Distance_ threshold, Output_& all_neighbors) const {
544 Distance_ threshold_raw = my_metric->denormalize(threshold);
545
546 /* Computing distances to all centers. We don't sort them here
547 * because the threshold is constant so there's no point.
548 */
549 Index_ ncenters = my_sizes.size();
550 auto center_ptr = my_centers.data();
551 for (Index_ center = 0; center < ncenters; ++center, center_ptr += my_dim) {
552 const Distance_ dist2center = my_metric->normalize(my_metric->raw(my_dim, target, center_ptr));
553
554 auto cur_nobs = my_sizes[center];
555 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
556 const Distance_ maxdist = *(dIt + cur_nobs - 1);
557
558 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
559 * threshold + maxdist < dist2center
560 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
561 */
562 const Distance_ lower_bd = dist2center - threshold;
563 if (maxdist < lower_bd) {
564 continue;
565 }
566
567 Index_ firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
568#if KNNCOLLE_KMKNN_USE_UPPER
569 /* This exploits the reverse triangle inequality, to ignore points where:
570 * threshold + dist2center < point-to-center distance
571 */
572 Distance_ upper_bd = threshold + dist2center;
573#endif
574
575 const auto cur_start = my_offsets[center];
576 auto other_ptr = my_data.data() + my_dim * static_cast<std::size_t>(cur_start + firstcell); // cast to avoid overflow issues.
577 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_ptr += my_dim) {
578#if KNNCOLLE_KMKNN_USE_UPPER
579 if (*(dIt + celldex) > upper_bd) {
580 break;
581 }
582#endif
583
584 auto dist2cell_raw = my_metric->raw(my_dim, target, other_ptr);
585 if (dist2cell_raw <= threshold_raw) {
586 if constexpr(count_only_) {
587 ++all_neighbors;
588 } else {
589 all_neighbors.emplace_back(dist2cell_raw, cur_start + celldex);
