355public:
359 template<class KmeansMatrix_ = kmeans::SimpleMatrix<Index_, Data_> >
360 KmknnPrebuilt(
361 size_t num_dim,
362 Index_ num_obs,
363 std::vector<Common<Data_, Distance_> > data,
364 std::shared_ptr<const DistanceMetric_> metric,
365 const KmknnOptions<Index_, Data_, Distance_, KmeansMatrix_>& options
366 ) :
367 my_dim(num_dim),
368 my_obs(num_obs),
369 my_data(std::move(data)),
370 my_metric(std::move(metric))
371 {
372 auto init = options.initialize_algorithm;
373 if (init == nullptr) {
374 init.reset(new InitializeKmeanspp<Index_, Data_, Distance_, KmeansMatrix_>);
375 }
376 auto refine = options.refine_algorithm;
377 if (refine == nullptr) {
378 refine.reset(new RefineHartiganWong<Index_, Data_, Distance_, KmeansMatrix_>);
379 }
380
381 Index_ ncenters = std::ceil(std::pow(my_obs, options.power));
382 my_centers.resize(static_cast<size_t>(ncenters) * my_dim); // cast to avoid overflow problems.
383
384 kmeans::SimpleMatrix<Index_, Data_> mat(my_dim, my_obs, my_data.data());
385 std::vector<Index_> clusters(my_obs);
386 auto output = kmeans::compute(mat, *init, *refine, ncenters, my_centers.data(), clusters.data());
387
388 // Removing empty clusters, e.g., due to duplicate points.
389 {
390 my_sizes.resize(ncenters);
391 std::vector<Index_> remap(ncenters);
392 Index_ survivors = 0;
393 for (Index_ c = 0; c < ncenters; ++c) {
394 if (output.sizes[c]) {
395 if (c > survivors) {
396 auto src = my_centers.begin() + static_cast<size_t>(c) * my_dim; // cast to avoid overflow.
397 auto dest = my_centers.begin() + static_cast<size_t>(survivors) * my_dim;
398 std::copy_n(src, my_dim, dest);
399 }
400 remap[c] = survivors;
401 my_sizes[survivors] = output.sizes[c];
402 ++survivors;
403 }
404 }
405
406 if (survivors < ncenters) {
407 for (auto& c : clusters) {
408 c = remap[c];
409 }
410 ncenters = survivors;
411 my_centers.resize(static_cast<size_t>(ncenters) * my_dim);
412 my_sizes.resize(ncenters);
413 }
414 }
415
416 my_offsets.resize(ncenters);
417 for (Index_ i = 1; i < ncenters; ++i) {
418 my_offsets[i] = my_offsets[i - 1] + my_sizes[i - 1];
419 }
420
421 // Organize points correctly; firstly, sorting by distance from the assigned center.
422 std::vector<std::pair<Distance_, Index_> > by_distance(my_obs);
423 {
424 auto sofar = my_offsets;
425 auto host = my_data.data();
426 for (Index_ o = 0; o < my_obs; ++o) {
427 auto optr = host + static_cast<size_t>(o) * my_dim;
428 auto clustid = clusters[o];
429 auto cptr = my_centers.data() + static_cast<size_t>(clustid) * my_dim;
430
431 auto& counter = sofar[clustid];
432 auto& current = by_distance[counter];
433 current.first = my_metric->normalize(my_metric->raw(my_dim, optr, cptr));
434 current.second = o;
435
436 ++counter;
437 }
438
439 for (Index_ c = 0; c < ncenters; ++c) {
440 auto begin = by_distance.begin() + my_offsets[c];
441 std::sort(begin, begin + my_sizes[c]);
442 }
443 }
444
445 // Permuting in-place to mirror the reordered distances, so that the search is more cache-friendly.
446 {
447 auto host = my_data.data();
448 std::vector<uint8_t> used(my_obs);
449 std::vector<Data_> buffer(my_dim);
450 my_observation_id.resize(my_obs);
451 my_dist_to_centroid.resize(my_obs);
452 my_new_location.resize(my_obs);
453
454 for (Index_ o = 0; o < my_obs; ++o) {
455 if (used[o]) {
456 continue;
457 }
458
459 const auto& current = by_distance[o];
460 my_observation_id[o] = current.second;
461 my_dist_to_centroid[o] = current.first;
462 my_new_location[current.second] = o;
463 if (current.second == o) {
464 continue;
465 }
466
467 // We recursively perform a "thread" of replacements until we
468 // are able to find the home of the originally replaced 'o'.
469 auto optr = host + static_cast<size_t>(o) * my_dim;
470 std::copy_n(optr, my_dim, buffer.begin());
471 Index_ replacement = current.second;
472 do {
473 auto rptr = host + static_cast<size_t>(replacement) * my_dim;
474 std::copy_n(rptr, my_dim, optr);
475 used[replacement] = 1;
476
477 const auto& next = by_distance[replacement];
478 my_observation_id[replacement] = next.second;
479 my_dist_to_centroid[replacement] = next.first;
480 my_new_location[next.second] = replacement;
481
482 optr = rptr;
483 replacement = next.second;
484 } while (replacement != o);
485
486 std::copy(buffer.begin(), buffer.end(), optr);
487 }
488 }
489
490 return;
491 }
496private:
497 void search_nn(const Common<Data_, Distance_>* target, knncolle::NeighborQueue<Index_, Distance_>& nearest, std::vector<std::pair<Distance_, Index_> >& center_order) const {
498 /* Computing distances to all centers and sorting them. The aim is to
499 * go through the nearest centers first, to try to get the shortest
500 * threshold (i.e., 'nearest.limit()') possible at the start;
501 * this allows us to skip searches of the later clusters.
502 */
503 center_order.clear();
504 size_t ncenters = my_sizes.size();
505 center_order.reserve(ncenters);
506 auto clust_ptr = my_centers.data();
507 for (size_t c = 0; c < ncenters; ++c, clust_ptr += my_dim) {
508 center_order.emplace_back(my_metric->raw(my_dim, target, clust_ptr), c);
509 }
510 std::sort(center_order.begin(), center_order.end());
511
512 // Computing the distance to each center, and deciding whether to proceed for each cluster.
513 Distance_ threshold_raw = std::numeric_limits<Distance_>::infinity();
514 for (const auto& curcent : center_order) {
515 const Index_ center = curcent.second;
516 const Distance_ dist2center = my_metric->normalize(curcent.first);
517
518 const auto cur_nobs = my_sizes[center];
519 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
520 const Distance_ maxdist = *(dIt + cur_nobs - 1);
521
522 Index_ firstcell = 0;
523#if KNNCOLLE_KMKNN_USE_UPPER
524 Distance_ upper_bd = std::numeric_limits<Distance_>::max();
525#endif
526
527 if (!std::isinf(threshold_raw)) {
528 const Distance_ threshold = my_metric->normalize(threshold_raw);
529
530 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
531 * threshold + maxdist < dist2center
532 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
533 */
534 const Distance_ lower_bd = dist2center - threshold;
535 if (maxdist < lower_bd) {
536 continue;
537 }
538
539 firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
540
541#if KNNCOLLE_KMKNN_USE_UPPER
542 /* This exploits the reverse triangle inequality, to ignore points where:
543 * threshold + dist2center < point-to-center distance
544 */
545 upper_bd = threshold + dist2center;
546#endif
547 }
548
549 const auto cur_start = my_offsets[center];
550 const auto* other_cell = my_data.data() + my_dim * static_cast<size_t>(cur_start + firstcell); // cast to avoid overflow issues.
551 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_cell += my_dim) {
552#if KNNCOLLE_KMKNN_USE_UPPER
553 if (*(dIt + celldex) > upper_bd) {
554 break;
555 }
556#endif
557
558 auto dist2cell_raw = my_metric->raw(my_dim, target, other_cell);
559 if (dist2cell_raw <= threshold_raw) {
560 nearest.add(cur_start + celldex, dist2cell_raw);
561 if (nearest.is_full()) {
562 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
563#if KNNCOLLE_KMKNN_USE_UPPER
564 upper_bd = my_metric->normalize(threshold_raw) + dist2center;
565#endif
566 }
567 }
568 }
569 }

572 template<bool count_only_, typename Output_>
573 void search_all(const Data_* target, Distance_ threshold, Output_& all_neighbors) const {
574 Distance_ threshold_raw = my_metric->denormalize(threshold);
575
576 /* Computing distances to all centers. We don't sort them here
577 * because the threshold is constant so there's no point.
578 */
579 Index_ ncenters = my_sizes.size();
580 auto center_ptr = my_centers.data();
581 for (Index_ center = 0; center < ncenters; ++center, center_ptr += my_dim) {
582 const Distance_ dist2center = my_metric->normalize(my_metric->raw(my_dim, target, center_ptr));
583
584 auto cur_nobs = my_sizes[center];
585 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
586 const Distance_ maxdist = *(dIt + cur_nobs - 1);
587
588 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
589 * threshold + maxdist < dist2center
590 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
591 */
592 const Distance_ lower_bd = dist2center - threshold;
593 if (maxdist < lower_bd) {
594 continue;
595 }
596
597 Index_ firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
598#if KNNCOLLE_KMKNN_USE_UPPER
599 /* This exploits the reverse triangle inequality, to ignore points where:
600 * threshold + dist2center < point-to-center distance
601 */
602 Distance_ upper_bd = threshold + dist2center;
603#endif
604
605 const auto cur_start = my_offsets[center];
606 auto other_ptr = my_data.data() + my_dim * static_cast<size_t>(cur_start + firstcell); // cast to avoid overflow issues.
607 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_ptr += my_dim) {
608#if KNNCOLLE_KMKNN_USE_UPPER
609 if (*(dIt + celldex) > upper_bd) {
610 break;
611 }
612#endif
613
614 auto dist2cell_raw = my_metric->raw(my_dim, target, other_ptr);
615 if (dist2cell_raw <= threshold_raw) {
616 if constexpr(count_only_) {
617 ++all_neighbors;
618 } else {
619 all_neighbors.emplace_back(dist2cell_raw, cur_start + celldex);
620 }
621 }
622 }
623 }
