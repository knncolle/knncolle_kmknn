189 template<class KmeansMatrix_ = kmeans::SimpleMatrix<Index_, Data_> >
190 KmknnPrebuilt(
191 std::size_t num_dim,
192 Index_ num_obs,
193 std::vector<Common<Data_, Distance_> > data,
194 std::shared_ptr<const DistanceMetric_> metric,
195 const KmknnOptions<Index_, Data_, Distance_, KmeansMatrix_>& options
196 ) :
197 my_dim(num_dim),
198 my_obs(num_obs),
199 my_data(std::move(data)),
200 my_metric(std::move(metric))
201 {
202 auto init = options.initialize_algorithm;
203 if (init == nullptr) {
204 init.reset(new InitializeKmeanspp<Index_, Data_, Distance_, KmeansMatrix_>);
205 }
206 auto refine = options.refine_algorithm;
207 if (refine == nullptr) {
208 refine.reset(new RefineHartiganWong<Index_, Data_, Distance_, KmeansMatrix_>);
209 }
210
211 const Index_ ncenters = std::ceil(std::pow(my_obs, options.power));
212 my_centers.resize(sanisizer::product<I<decltype(my_centers.size())> >(ncenters, my_dim));
213
214 kmeans::SimpleMatrix<Index_, Data_> mat(my_dim, my_obs, my_data.data());
215 auto clusters = sanisizer::create<std::vector<Index_> >(my_obs);
216 auto output = kmeans::compute(mat, *init, *refine, ncenters, my_centers.data(), clusters.data());
217
218 // Removing empty clusters, e.g., due to duplicate points.
219 {
220 sanisizer::resize(my_sizes, ncenters);
221 auto remap = sanisizer::create<std::vector<Index_> >(ncenters);
222 Index_ survivors = 0;
223 for (Index_ c = 0; c < ncenters; ++c) {
224 if (output.sizes[c]) {
225 if (c > survivors) {
226 auto src = my_centers.begin() + sanisizer::product_unsafe<std::size_t>(c, my_dim);
227 auto dest = my_centers.begin() + sanisizer::product_unsafe<std::size_t>(survivors, my_dim);
228 std::copy_n(src, my_dim, dest);
229 }
230 remap[c] = survivors;
231 my_sizes[survivors] = output.sizes[c];
232 ++survivors;
233 }
234 }
235
236 if (survivors < ncenters) {
237 for (auto& c : clusters) {
238 c = remap[c];
239 }
240 ncenters = survivors; // this should be safe as survivors is always smaller than ncenters.
241 my_centers.resize(sanisizer::product_unsafe<I<decltype(my_centers.size())> >(ncenters, my_dim));
242 my_sizes.resize(ncenters);
243 }
244 }
245
246 sanisizer::resize(my_offsets, ncenters);
247 for (Index_ i = 1; i < ncenters; ++i) {
248 my_offsets[i] = my_offsets[i - 1] + my_sizes[i - 1];
249 }
250
251 // Organize points correctly; firstly, sorting by distance from the assigned center.
252 auto by_distance = sanisizer::create<std::vector<std::pair<Distance_, Index_> > >(my_obs);
253 {
254 auto sofar = my_offsets;
255 auto host = my_data.data();
256 for (Index_ o = 0; o < my_obs; ++o) {
257 auto optr = host + sanisizer::product_unsafe<std::size_t>(o, my_dim);
258 auto clustid = clusters[o];
259 auto cptr = my_centers.data() + sanisizer::product_unsafe<std::size_t>(clustid, my_dim);
260
261 auto& counter = sofar[clustid];
262 auto& current = by_distance[counter];
263 current.first = my_metric->normalize(my_metric->raw(my_dim, optr, cptr));
264 current.second = o;
265
266 ++counter;
267 }
268
269 for (Index_ c = 0; c < ncenters; ++c) {
270 auto begin = by_distance.begin() + my_offsets[c];
271 std::sort(begin, begin + my_sizes[c]);
272 }
273 }
274
275 // Permuting in-place to mirror the reordered distances, so that the search is more cache-friendly.
276 {
277 auto host = my_data.data();
278 auto used = sanisizer::create<std::vector<unsigned char> >(my_obs);
279 auto buffer = sanisizer::create<std::vector<Data_> >(my_dim);
280 sanisizer::resize(my_observation_id, my_obs);
281 sanisizer::resize(my_dist_to_centroid, my_obs);
282 sanisizer::resize(my_new_location, my_obs);
283
284 for (Index_ o = 0; o < my_obs; ++o) {
285 if (used[o]) {
286 continue;
287 }
288
289 const auto& current = by_distance[o];
290 my_observation_id[o] = current.second;
291 my_dist_to_centroid[o] = current.first;
292 my_new_location[current.second] = o;
293 if (current.second == o) {
294 continue;
295 }
296
297 // We recursively perform a "thread" of replacements until we
298 // are able to find the home of the originally replaced 'o'.
299 auto optr = host + sanisizer::product_unsafe<std::size_t>(o, my_dim);
300 std::copy_n(optr, my_dim, buffer.begin());
301 Index_ replacement = current.second;
302 do {
303 auto rptr = host + sanisizer::product_unsafe<std::size_t>(replacement, my_dim);
304 std::copy_n(rptr, my_dim, optr);
305 used[replacement] = 1;
306
307 const auto& next = by_distance[replacement];
308 my_observation_id[replacement] = next.second;
309 my_dist_to_centroid[replacement] = next.first;
310 my_new_location[next.second] = replacement;
311
312 optr = rptr;
313 replacement = next.second;
314 } while (replacement != o);
315
316 std::copy(buffer.begin(), buffer.end(), optr);
317 }
318 }
319
320 return;
321 }

324 void search_nn(const Common<Data_, Distance_>* target, knncolle::NeighborQueue<Index_, Distance_>& nearest, std::vector<std::pair<Distance_, Index_> >& center_order) const {
325 /* Computing distances to all centers and sorting them. The aim is to
326 * go through the nearest centers first, to try to get the shortest
327 * threshold (i.e., 'nearest.limit()') possible at the start;
328 * this allows us to skip searches of the later clusters.
329 */
330 center_order.clear();
331 const auto ncenters = my_sizes.size();
332 center_order.reserve(ncenters);
333 for (decltype(ncenters) c = 0; c < ncenters; ++c) {
334 auto clust_ptr = my_centers.data() + sanisizer::product_unsafe<std::size_t>(c, my_dim);
335 center_order.emplace_back(my_metric->raw(my_dim, target, clust_ptr), c);
336 }
337 std::sort(center_order.begin(), center_order.end());
338
339 // Computing the distance to each center, and deciding whether to proceed for each cluster.
340 Distance_ threshold_raw = std::numeric_limits<Distance_>::infinity();
341 for (const auto& curcent : center_order) {
342 const Index_ center = curcent.second;
343 const Distance_ dist2center = my_metric->normalize(curcent.first);
344
345 const auto cur_nobs = my_sizes[center];
346 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
347 const Distance_ maxdist = *(dIt + cur_nobs - 1);
348
349 Index_ firstcell = 0;
350#if KNNCOLLE_KMKNN_USE_UPPER
351 Distance_ upper_bd = std::numeric_limits<Distance_>::max();
352#endif
353
354 if (!std::isinf(threshold_raw)) {
355 const Distance_ threshold = my_metric->normalize(threshold_raw);
356
357 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
358 * threshold + maxdist < dist2center
359 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
360 */
361 const Distance_ lower_bd = dist2center - threshold;
362 if (maxdist < lower_bd) {
363 continue;
364 }
365
366 firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
367
368#if KNNCOLLE_KMKNN_USE_UPPER
369 /* This exploits the reverse triangle inequality, to ignore points where:
370 * threshold + dist2center < point-to-center distance
371 */
372 upper_bd = threshold + dist2center;
373#endif
374 }
375
376 const auto cur_start = my_offsets[center];
377 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex) {
378 const auto other_cell = my_data.data() + sanisizer::product_unsafe<std::size_t>(cur_start + celldex, my_dim);
379
380#if KNNCOLLE_KMKNN_USE_UPPER
381 if (*(dIt + celldex) > upper_bd) {
382 break;
383 }
384#endif
385
386 auto dist2cell_raw = my_metric->raw(my_dim, target, other_cell);
387 if (dist2cell_raw <= threshold_raw) {
388 nearest.add(cur_start + celldex, dist2cell_raw);
389 if (nearest.is_full()) {
390 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
391#if KNNCOLLE_KMKNN_USE_UPPER
392 upper_bd = my_metric->normalize(threshold_raw) + dist2center;
393#endif
394 }
395 }
396 }
397 }

400 template<bool count_only_, typename Output_>
401 void search_all(const Data_* target, Distance_ threshold, Output_& all_neighbors) const {
402 Distance_ threshold_raw = my_metric->denormalize(threshold);
403
404 /* Computing distances to all centers. We don't sort them here
405 * because the threshold is constant so there's no point.
406 */
407 const auto ncenters = my_sizes.size();
408 for (I<decltype(ncenters)> center = 0; center < ncenters; ++center) {
409 auto center_ptr = my_centers.data() + sanisizer::product_unsafe<std::size_t>(center, my_dim);
410 const Distance_ dist2center = my_metric->normalize(my_metric->raw(my_dim, target, center_ptr));
411
412 auto cur_nobs = my_sizes[center];
413 const Distance_* dIt = my_dist_to_centroid.data() + my_offsets[center];
414 const Distance_ maxdist = *(dIt + cur_nobs - 1);
415
416 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
417 * threshold + maxdist < dist2center
418 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
419 */
420 const Distance_ lower_bd = dist2center - threshold;
421 if (maxdist < lower_bd) {
422 continue;
423 }
424
425 Index_ firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
426#if KNNCOLLE_KMKNN_USE_UPPER
427 /* This exploits the reverse triangle inequality, to ignore points where:
428 * threshold + dist2center < point-to-center distance
429 */
430 Distance_ upper_bd = threshold + dist2center;
431#endif
432
433 const auto cur_start = my_offsets[center];
434 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex) {
435 const auto other_ptr = my_data.data() + sanisizer::product_unsafe<std::size_t>(cur_start + celldex, my_dim);
436
437#if KNNCOLLE_KMKNN_USE_UPPER
438 if (*(dIt + celldex) > upper_bd) {
439 break;
440 }
441#endif
442
443 auto dist2cell_raw = my_metric->raw(my_dim, target, other_ptr);
444 if (dist2cell_raw <= threshold_raw) {
445 if constexpr(count_only_) {
446 ++all_neighbors;
447 } else {
448 all_neighbors.emplace_back(dist2cell_raw, cur_start + celldex);
449 }
450 }
451 }
452 }

620 auto build_known_raw(const Matrix_& data) const {
621 const auto ndim = data.num_dimensions();
622 const auto nobs = data.num_observations();
623
624 typedef std::vector<Common<Data_, Distance_> > Store;
625 Store store(sanisizer::product<typename Store::size_type>(ndim, nobs));

628 for (I<decltype(nobs)> o = 0; o < nobs; ++o) {
629 auto ptr = work->next();
630 std::copy_n(ptr, ndim, store.begin() + sanisizer::product_unsafe<std::size_t>(o, ndim));
631 }
